每日一题，用ListNode实现链表

2022/9/23 算法 C++ 力扣 链表

实现题

get(index)：获取链表中第 index 个节点的值。如果索引无效，则返回-1。
addAtHead(val)：在链表的第一个元素之前添加一个值为 val 的节点。插入后，新节点将成为链表的第一个节点。
addAtTail(val)：将值为 val 的节点追加到链表的最后一个元素。
addAtIndex(index,val)：在链表中的第 index 个节点之前添加值为 val  的节点。如果 index 等于链表的长度，则该节点将附加到链表的末尾。如果 index 大于链表长度，则不会插入节点。如果index小于0，则在头部插入节点。
deleteAtIndex(index)：如果索引 index 有效，则删除链表中的第 index 个节点。

单链表中的节点应该具有两个属性：
val 和 next。val 是当前节点的值，
next 是指向下一个节点的指针/引用。
如果要使用双向链表，
则还需要一个属性 prev 以指示链表中的上一个节点。
假设链表中的所有节点都是 0-index 的。

我的代码：

class MyLinkedList {
private:
    int size;
    ListNode *head;
public:
    MyLinkedList() {
        this->size = 0;
        this->head = new ListNode(0);

    }
    
    int get(int index) {
        if(index<0 || index >=size){
            return -1;
        }
        else{
            ListNode *cur=head;
            for(int i=0;i<=index;i++){
                cur=cur->next;
            }
            return cur->val;
        }
    }
    
    void addAtHead(int val) {
        addAtIndex(0,val);
    }
    
    void addAtTail(int val) {
        addAtIndex(size,val);
    }
    
    void addAtIndex(int index, int val) {
        if(index>size||index<0){
           return;
        }
        else{
            size++;
            ListNode *cur=head;
            ListNode *preCur=new ListNode(val);//节点赋值
            for(int i=0;i<index;i++){
                cur=cur->next;
            }//循环到index前一位
            preCur->next=cur->next;//preCur->index (preCur是index位置,index变成index+1)
            cur->next=preCur;//index-1 ->preCur ->index + 1
        }
    }
    
    void deleteAtIndex(int index) {
        if(index<0||index>=size){
            return;
        }
        size--;
        ListNode *cur=head;
         for(int i=0;i<index;i++){
                cur=cur->next;
        }//到达i-1
        ListNode *n=cur->next;
        cur->next=cur->next->next;
        delete n;
        //n只是用来删除的？
        //cur(i-1)->index+1
    }
};
