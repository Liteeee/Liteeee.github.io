代码示例

import torch

import flash
from flash.audio import AudioClassificationData
from flash.core.data.utils import download_data
from flash.image import ImageClassifier

# 1. Create the DataModule
download_data("https://pl-flash-data.s3.amazonaws.com/urban8k_images.zip", "./data")

datamodule = AudioClassificationData.from_folders(
    train_folder="data/urban8k_images/train",
    val_folder="data/urban8k_images/val",
    transform_kwargs=dict(spectrogram_size=(64, 64)),
    batch_size=4,
)

# 2. Build the model.
model = ImageClassifier(backbone="resnet18", labels=datamodule.labels)

# 3. Create the trainer and finetune the model
trainer = flash.Trainer(max_epochs=3, gpus=torch.cuda.device_count())
trainer.finetune(model, datamodule=datamodule, strategy=("freeze_unfreeze", 1))

# 4. Predict what's on few images! air_conditioner, children_playing, siren etc.
datamodule = AudioClassificationData.from_files(
    predict_files=[
        "data/urban8k_images/test/air_conditioner/13230-0-0-5.wav.jpg",
        "data/urban8k_images/test/children_playing/9223-2-0-15.wav.jpg",
        "data/urban8k_images/test/jackhammer/22883-7-10-0.wav.jpg",
    ],
    batch_size=3,
)
predictions = trainer.predict(model, datamodule=datamodule, output="labels")
print(predictions)

# 5. Save the model!
trainer.save_checkpoint("audio_classification_model.pt")

import librosa.display
import numpy as np
import matplotlib.pyplot as plt
import os
from pydub import AudioSegment
from pathlib import Path
from datetime import datetime

class ProTools:
    """项目工具"""
    def __init__(self):
        print("实例化")
    #拼接音频
    def toImage(self,path,output_path,date):
        """wav音频转jpg频谱图"""
        ##path = "./data/cos/test/healthy/cough-heavy4.wav"

        # sr=None声音保持原采样频率， mono=False声音保持原通道数
        data, fs = librosa.load(path, sr=None, mono=False)

        L = len(data)
        print('Time:', L / fs)

        # 0.025s
        framelength = 0.025
        # NFFT点数=0.025*fs
        framesize = int(framelength * fs)
        print("NFFT:", framesize)

        # 提取mel特征
        mel_spect = librosa.feature.melspectrogram(data, sr=fs, n_fft=framesize)
        # 转化为log形式
        mel_spect = librosa.power_to_db(mel_spect, ref=np.max)

        # 画mel谱图
        librosa.display.specshow(mel_spect, sr=fs, x_axis='time', y_axis='mel')

        #保存路径
        plt.savefig(output_path+date+".jpg")
        #路径应该是目录+动态文件名
        #先保存再show，不然就会是空白
        plt.show()

    #合并音頻
    def SpliceAudio(self,path,outputpath,id):
        ##path = "test"
        date = datetime.today().strftime("%Y%m%d%H%M%S%f")  # 生成日期
        files = [file for file in Path(path).iterdir() if file.name.endswith(".wav")]  # 遍历sounds目录下的文件

        def merge_wav(files):
            song = AudioSegment.silent(duration=100)
            for file in sorted(files, reverse=True):
                songclip = AudioSegment.from_file(file)
                song = songclip + song
            song = AudioSegment.silent(duration=2000) + song
            return song

        sound = merge_wav(files)  # 合并mp3
        sound.export(outputpath+'/'+f'{date}.wav', format("wav"))  # 把合并合的mp3进行合成然后重命名。
        #合并完成后，调用生成图片的函数
        self.toImage(outputpath+'/'+f'{date}.wav',outputpath,date)
        # 调用生成函数之后把刚刚生成的文件删掉，不然反复运行时会出错
        os.remove(outputpath+'/'+f'{date}.wav')

    def GetWavList(self,path):
        # path定义要获取的文件名称的目录
        # os.listdir()方法获取文件夹名字，返回数组
        file_name_list = os.listdir(path)
        res_list=[]
        for file in file_name_list:
            #去掉json文件
            if(file.find(".json")==-1):
                res_list.append(path+"/"+file)
        print(res_list)
        return res_list

    def GetAllFilesPath(self,path):
        list=[]
        label_floader = [file for file in Path(path).iterdir()]  # 遍历sounds目录下的文件
        #分类路径
        for label_path in label_floader:
            #print(type(label_path))
            #每个测试者ID路径
            id_floaders = [file for file in Path(label_path).iterdir()]
            #遍历测试者
            for id_path in id_floaders:
                print(id_path)
                id=1
                #输入PATH是用户ID,输出PATH是分类PATH
                self.SpliceAudio(id_path,str(id_path),id)
        return list
