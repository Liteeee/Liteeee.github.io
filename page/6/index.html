def cal_acf(x, nlags):
    x = np.array(x)
    mean_x = np.mean(x)#求均值ex
    print('ex:',mean_x)
    length_x = x.shape[0]#序列长度
    c_0 = np.mean((x-mean_x) **2)#E[（x-E(x)）^2]
    print('c_0:',c_0)
    c_k = np.sum((x[:(length_x-nlags)] - mean_x) * (x[nlags:] - mean_x)) / length_x
    #从第零个元素到x-k个元素 x[:(length_x-nlags)]=记为Xi~Xk
    #所以[X1~Xk(保留前k项目)-E(X)]*[Xk~Xn（删除前k项）-E（x)] / T
    #累加len-k次

此处对应

    r_k = c_k / c_0
    return r_k
